Re: Copy Specific File from Network Folder

Code: Select all

@echo off
set/p  "moth=Enter month in mm/dd/yy format: "
set "month=%moth:~0,2%
set day=%moth:~3,-3%
set year=%moth:~6%
set "file=ABC%month%%day%%year%X.zip"
copy %file% D:\CopyFile 2>nul
if %ERRORLEVEL% GTR 0 (echo Specified File Not Found!)
pause
