Hi,

nearly the same like :isNumeric.

::************************
:isAlphaNumeric
setlocal
set rest=_
for /f "delims=0123456789abcdefghijklmnopqrstuvwxyz" %%a in ("%~1") do set rest=_%%a
set return=false
if "%rest%"=="_" set return=true
( endlocal
if "%~2" NEQ "" set %~2=%return%
goto:eof
)
