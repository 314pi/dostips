Code: Select all

@echo off
rem For each folder in your folders
for /R %%~dp0 %%i in (.) do (
cd "%%~dp/%%i"
if "%%~xa" NEQ ""  if "%%~dpxi" NEQ "%~dpx0" (
rem Move the file to directory
move /y "%%i/*" ../
))

You mean so?

Code: Select all

PushD D:\rootfolder
For /r %%i in (*) do if "%%~fi" neq "%~f0" if NOT exist %%~nxi move "%%i"
Popd
