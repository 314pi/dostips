Code: Select all

@echo off
setlocal EnableDelayedExpansion

rem Note that this is a TAB: ---v
(for /F "tokens=1,2,3,7 delims=/	 " %%a in (input.txt) do (
   set /A "YMD=100000000-(%%c*10000+%%a*100+%%b)"
   echo !YMD! %%d
)) > temp.txt

set "i=0"
for /F "tokens=2" %%a in ('sort temp.txt') do (
   set /A i+=1
   if !i! leq 2 echo %%a
)
del temp.txt
